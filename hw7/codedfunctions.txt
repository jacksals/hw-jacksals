bool find(value, root)
{
	if(root == NULL)
	{
		return false;
	}	
	if(root->value == value)
	{
		return true;
	}
	if(value > root->value)
	{
		return find(value, root->right);
	}
	else
	{
		return find(value, root->left);
	}
}
